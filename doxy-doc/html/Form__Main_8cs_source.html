409  private void StartUpBothModes1And2(int mode2, string startingdir)
410  {
411  //Erase the datamodel and stuff. Very important.
412  PrepareDataModelForUse();
413 
414  switch (mode2)
415  {
416  case (0):
417  if (_dataModel.fixreadonlyActive > 0)
418  _dataModel.FixReadonlyResults();
419 
420  if (Settings.Default.mode1addrootdir)
421  {
422  DateTime? nullorfileDateTime = _dataModel.DecideWhichTimeMode1(startingdir);
423  if (nullorfileDateTime == null) //if nothing could be decided, exit, otherwise continue
424  {
425  MessageBox.Show("Error! Nothing to decide time from. \n" +
426  "Sub-File or Sub-Dir button requires files/dirs to be SHOWN on the right-side Contents panel...",
427  @"PEBKAC Error ", MessageBoxButtons.OK, MessageBoxIcon.Error);
428  return;
429  }
430 
431  _dataModel.AddONEFiletoConfirmList(startingdir, (DateTime)nullorfileDateTime, true);
432  }
433 
434  //TODO: where I should add worker process
435  if (Settings.Default.useRootDirAsContainer)
436  {
437  try
438  {
439  foreach (string subfolder in Directory.GetDirectories(startingdir))
440  _dataModel.RecurseSubDirectoryMode1(Path.Combine(startingdir, subfolder));
441  }
442  catch (UnauthorizedAccessException)
443  { }
444  catch (DirectoryNotFoundException)
445  { }
446  } else
447  _dataModel.RecurseSubDirectoryMode1(startingdir);
448  //end worker process
449  break;
450 
451  case (1):
452  //TODO: here too:
453  if (Settings.Default.useRootDirAsContainer)
454  {
455  try
456  {
457  foreach (string subfolder in Directory.GetDirectories(startingdir))
458  _dataModel.RecurseSubDirectoryMode2(Path.Combine(startingdir, subfolder));
459  }
460  catch (UnauthorizedAccessException)
461  { }
462  catch (DirectoryNotFoundException)
463  { }
464  } else
465  _dataModel.RecurseSubDirectoryMode2(startingdir);

469  //TODO: THREEDO: combine both.
470  }
