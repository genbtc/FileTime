351  private void StartUpBothModes1And2(int mode2, string startingdir)
352  {
353  _dataModel.FilestoConfirmList.Clear();
354 
355  _dataModel.Skips.Reset();
356 
357  _dataModel.gui = GetGUIRadioButtonStatusData();
358 
359  _dataModel.checkboxes = QueryCMAcheckboxes();
360 
361  switch (mode2)
362  {
363  case (0):
364  if (Confirmation.active > 0)
365  Confirmation.FixReadonlyResults();
366 
367  if (Settings.Default.mode1addrootdir)
368  {
369  DateTime? nullorfileDateTime = _dataModel.DecideWhichTimeMode1(startingdir);
370  if (nullorfileDateTime == null) //if nothing could be decided, exit, otherwise continue
371  {
372  MessageBox.Show("Error! Nothing to decide time from. \n" +
373  "Sub-File or Sub-Dir button requires files/dirs to be SHOWN on the right-side Contents panel...",
374  "PEBKAC Error ", MessageBoxButtons.OK, MessageBoxIcon.Error);
375  return;
376  }
377  var fileDateTime = (DateTime)nullorfileDateTime;
378 
379  _dataModel.SetFileDateTimeMode1(startingdir, fileDateTime, true);
380  }
381 
382  //TODO: where I should add worker process
383  if (Settings.Default.useRootDirAsContainer)
384  _dataModel.RecurseSubDirectoryMode1Parent(startingdir, checkBox_Recurse.Checked, checkBoxShouldFiles.Checked);
385  else
386  _dataModel.RecurseSubDirectoryMode1(startingdir, checkBox_Recurse.Checked, checkBoxShouldFiles.Checked);
387  //end worker process
388  break;
389 
390  case (1):
391  if (Settings.Default.useRootDirAsContainer)
392  _dataModel.RecurseSubDirectoryMode2Parent(startingdir);
393  else
394  _dataModel.RecurseSubDirectoryMode2(startingdir);
395  break;
396  }
397 
398  var itemsSkippedCount = _dataModel.Skips.H + _dataModel.Skips.R + _dataModel.Skips.S;
399  if (itemsSkippedCount > 0)
400  {
401  string skippedmessage = "";
402  skippedmessage += "There were " + _dataModel.Skips.S + " System files/directories skipped.\n";
403  skippedmessage += "There were " + _dataModel.Skips.H + " Hidden files/directories skipped.\n";
404  skippedmessage += "There were " + _dataModel.Skips.R + " Read-Only files/directories skipped.";
405  MessageBox.Show(skippedmessage, "Info Log", MessageBoxButtons.OK, MessageBoxIcon.Information);
406  }
407  //Launch Form 2
408  LaunchForm2Go();
409  }
