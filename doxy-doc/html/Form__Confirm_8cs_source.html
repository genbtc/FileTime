119  private static bool DoConditionalAgeCheck(int cmaType, string pathName, DateTime newDate, bool useNewer, bool useOlder)
120  {
121  try
122  {
123  var olddate = new DateTime();

125  olddate = File.GetCreationTime(pathName);

127  olddate = File.GetLastWriteTime(pathName);
128  if (cmaType == 3)
129  olddate = File.GetLastAccessTime(pathName);
