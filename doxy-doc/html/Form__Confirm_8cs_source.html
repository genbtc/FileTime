167  private bool DoConditionalAgeCheck(DateTime newdate, string pathName, int cmAtype)
168  {
169  try

172  if (cmAtype == 1)

176  if (cmAtype == 3)
177  olddate = File.GetLastAccessTime(pathName);
178  if (radioButton1Newer.Checked)
