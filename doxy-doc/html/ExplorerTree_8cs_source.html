524  public void BrowseTo()
525  {
526  Cursor.Current = Cursors.WaitCursor;
527  try
528  {
529  PopulateMyComputer();
530  string textboxPath = txtPath.Text.ToLower();
531 
532  int sentinel = 1;
533  TreeNode loopRootNode = nodeMyComputer;
534  bool network = false;
535  if (textboxPath.Contains(Desktop.ToLower()))
536  {
537  loopRootNode = rootnodeDesktop;
538  if (textboxPath.Length == Desktop.Length)
539  {
540  loopRootNode.Expand();
541  loopRootNode.EnsureVisible();
542  return;
543  }
544  }
545  else if (textboxPath.Contains(MyDocuments.ToLower()))
546  {
547  loopRootNode = nodeMyDocuments;
548  if (textboxPath.Length == MyDocuments.Length)
549  {
550  loopRootNode.Expand();
551  loopRootNode.EnsureVisible();
552  return;
553  }
554  }
555  else if (textboxPath == "my computer")
556  {
557  loopRootNode.Expand();
558  loopRootNode.EnsureVisible();
559  return;
560  }
561  else if (textboxPath.StartsWith(@"\\"))
562  {
563  loopRootNode = nodeEntireNetwork;
564  network = true;
565  }
566  treeview1.SelectedNode = loopRootNode;
567  if (!textboxPath.EndsWith(Seperator))
568  textboxPath += Seperator;
569 
570  StartAgain:
571 
572  do
573  {
574  foreach (TreeNode currNode in loopRootNode.Nodes)
575  {
576  string currNodePath = currNode.Tag.ToString().ToLower();
577  if (!currNodePath.EndsWith(Seperator))
578  currNodePath += Seperator;
579 
580  if (textboxPath.StartsWith(currNodePath))
581  {
582  currNode.TreeView.Focus();
583  currNode.TreeView.SelectedNode = currNode;
584  //currNode.EnsureVisible();
585  NativeScroll.ScrollH(currNode, 37);
586  if (currNode.Nodes.Count > 0)
587  {
588  currNode.Expand();
589  loopRootNode = currNode;
590  }
591  else
592  {
593  if (textboxPath == currNodePath)
594  {
595  sentinel = -1;
596  break;
597  }
598  continue;
599  }
600  if (currNodePath.StartsWith(textboxPath))
601  {
602  sentinel = -1;
603  break;
604  }
605  goto StartAgain;
606  }
607  if (network)
608  {
609  loopRootNode = currNode;
610  goto StartAgain;
611  }
612  }
613  if (sentinel == -1)
614  break;
615  try
616  {
617  loopRootNode = loopRootNode.NextNode;
618  }
619  catch (Exception)
620  { } //no more nodes (not an error, so suppress)
621  } while (sentinel >= 0);
622  }
623  catch (Exception e1)
624  {
625  MessageBox.Show("Error: " + e1.Message);
626  }
627  Cursor.Current = Cursors.Default;
628  }

922  private void EnumerateServers(NETRESOURCE pRsrc, ResourceScope scope, ResourceType type, ResourceUsage usage,
923  ResourceDisplayType displayType, string kPath)
924  {
925  uint bufferSize = 16384;
926  IntPtr buffer = Marshal.AllocHGlobal((int)bufferSize);
927  IntPtr handle;
928  uint cEntries = 1;
929  bool serverenum = false;
930 
931  ErrorCodes result = WNetOpenEnum(scope, type, usage, pRsrc, out handle);
932 
933  if (result == ErrorCodes.NO_ERROR)
934  {
935  do
936  {
937  result = WNetEnumResource(handle, ref cEntries, buffer, ref bufferSize);
938 
939  if ((result == ErrorCodes.NO_ERROR))
940  {
941  Marshal.PtrToStructure(buffer, pRsrc);
942 
943  if (kPath == "")
944  {
945  if ((pRsrc.dwDisplayType == displayType) ||
946  (pRsrc.dwDisplayType == ResourceDisplayType.RESOURCEDISPLAYTYPE_DOMAIN))
947  aData.Add(pRsrc.lpRemoteName + "|" + pRsrc.dwDisplayType);
948 
949  if ((pRsrc.dwUsage & ResourceUsage.RESOURCEUSAGE_CONTAINER) ==
950  ResourceUsage.RESOURCEUSAGE_CONTAINER)
951  {
952  if ((pRsrc.dwDisplayType == displayType))
953  EnumerateServers(pRsrc, scope, type, usage, displayType, kPath);
954  }
955  }
956  else
957  {
958  if (pRsrc.dwDisplayType == displayType)
959  {
960  aData.Add(pRsrc.lpRemoteName);
961  EnumerateServers(pRsrc, scope, type, usage, displayType, kPath);
962  //return;
963  serverenum = true;
964  }
965  if (!serverenum)
966  {
967  if (pRsrc.dwDisplayType == ResourceDisplayType.RESOURCEDISPLAYTYPE_SHARE)
968  aData.Add(pRsrc.lpRemoteName + "_share");
969  }
970  else
971  serverenum = false;
972  if ((kPath.IndexOf(pRsrc.lpRemoteName) >= 0) ||
973  (pRsrc.lpRemoteName == "Microsoft Windows Network"))
974  EnumerateServers(pRsrc, scope, type, usage, displayType, kPath);
975  }
976  }
977  else if (result != ErrorCodes.ERROR_NO_MORE_ITEMS)
978  break;
979  } while (result != ErrorCodes.ERROR_NO_MORE_ITEMS);
980 
981  WNetCloseEnum(handle);
982  }
983 
984  Marshal.FreeHGlobal(buffer);
985  }
